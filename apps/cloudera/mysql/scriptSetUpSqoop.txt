CREATE USER 'demo' IDENTIFIED BY  '1234';

CREATE DATABASE demo;

GRANT ALL PRIVILEGES ON demo.* TO 'demo';

CREATE TABLE IF NOT EXISTS demo1.posts (
    id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT,
    postTypeId INT,
	creationDate LONG,
	score INT,
	viewCount INT,
	body VARCHAR(255),
	ownerUserId INT,
	lastActivityDate LONG,
	title VARCHAR(255),
	answerCount INT,
	commentCount INT,
	favoriteCount INT,
	tags VARCHAR(255),
    PRIMARY KEY (id)
);
